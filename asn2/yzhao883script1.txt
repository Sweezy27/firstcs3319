-- create the patient, hospital, doctor and treating table
SHOW TABLES;

CREATE TABLE patient (ohip CHAR(9) NOT NULL, firstname VARCHAR(20), lastname VARCHAR(20), PRIMARY KEY (ohip));

CREATE TABLE hospital (code CHAR(3) NOT NULL, name VARCHAR(20), city VARCHAR(20), province CHAR(2), numofbeds VARCHAR(8), head CHAR(4), dateofheadstart DATE, PRIMARY KEY (code));

CREATE TABLE doctor (firstname VARCHAR(20), lastname VARCHAR(20), licensenum CHAR(4) NOT NULL, datebelicensed DATE, specialty VARCHAR(30), workoutof CHAR(3), PRIMARY KEY (licensenum), FOREIGN KEY (workoutof) REFERENCES hospital(code)); 

CREATE TABLE treating (licensenum CHAR(4) NOT NULL, OHIP CHAR(9) NOT NULL, FOREIGN KEY (licensenum) REFERENCES doctor(licensenum) ON DELETE CASCADE, FOREIGN KEY (ohip) REFERENCES patient(ohip), PRIMARY KEY (licensenum, ohip));
SHOW TABLES;

-- add foreign key in hospital reference to doctor
ALTER TABLE hospital ADD FOREIGN KEY (head) REFERENCES doctor(licensenum);
DESCRIBE hospital;
